import shodan

API_KEY = "YOUR_API_KEY"
api = shodan.Shodan(API_KEY)

try:
    results = api.search('apache')

    print(f"Results found: {results['total']}")
    for result in results['matches']:
        print(f"IP: {result['ip_str']}")
        print(result['data'])
        print("-" * 50)
except Exception as e:
    print(f"Error: {e}")
